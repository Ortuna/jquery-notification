jQuery notification

A simple and small notification plugin for jQuery

Demo

Usage

$.createNotification({
	click: function() {},	// Click callback
	content: '',		// Content
	duration: 5000,		// On screen duration in milliseconds
	fadeIn: 400,		// Fade in in milliseconds
	fadeOut: 400,		// Fade out in milliseconds
	limit: false,		// Do not add if already reached limit
	queue: false,		// If reached limited and queue is active add when a slot is free
	slideUp: 200,		// Slide up in milliseconds
	horizontal: 'right',	// Horizontal align
	vertical: 'top'		// Vertical align
});
